cmake_minimum_required(VERSION 3.30)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

include(FetchContent)

# Declare project
project(EntropyTest VERSION 1.0.0 LANGUAGES CXX)

# Fetch GoogleTest
FetchContent_Declare(
        googletest
        URL https://github.com/google/googletest/archive/refs/tags/v1.14.0.zip
)
FetchContent_MakeAvailable(googletest)

add_subdirectory(.. Entropy)

# Glob source files
file(GLOB_RECURSE TEST_SOURCES ${PROJECT_SOURCE_DIR}/unit/*.cpp)

add_executable(${PROJECT_NAME} ${TEST_SOURCES})

target_include_directories(${PROJECT_NAME} PRIVATE ${ENTROPY_INCLUDE_DIRS})
target_link_libraries(${PROJECT_NAME} PRIVATE gtest gtest_main glfw Entropy glm quill GPUOpen::VulkanMemoryAllocator flecs::flecs_static)
